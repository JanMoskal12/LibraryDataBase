CREATE DATABASE Biblioteka;
USE Biblioteka;

CREATE TABLE Półka
(
Id INTEGER PRIMARY KEY,
Tytuł VARCHAR(20) NOT NULL,
Gatunek VARCHAR(20) NOT NULL,
Autor VARCHAR(40) NOT NULL,
Rok_wydania CHAR(4),
Wydawnictwo VARCHAR(15)
);

CREATE TABLE Rezerwacje
(
Id INTEGER PRIMARY KEY,
Id_książki INTEGER,
Id_klienta INTEGER,
CONSTRAINT Id_książki_fk FOREIGN KEY (Id_książki) REFERENCES Półka(Id)
);

CREATE TABLE Klienci
(
Id INTEGER PRIMARY KEY,
Imie VARCHAR(20) NOT NULL,
Nazwisko VARCHAR(30) NOT NULL,
NumerTelefonu CHAR(9) NOT NULL);
