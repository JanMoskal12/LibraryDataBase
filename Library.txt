CREATE DATABASE Biblioteka;
USE Biblioteka;
CREATE TABLE Półka
(
Id INTEGER PRIMARY KEY AUTO_INCREMENT,
Tytuł VARCHAR(20) NOT NULL,
Gatunek VARCHAR(20) NOT NULL,
Autor VARCHAR(30) NOT NULL,
Rok_wydania CHAR(4),
Wydawnictwo VARCHAR(15)
);
CREATE TABLE Rezerwacje
(
Id INTEGER PRIMARY KEY AUTO_INCREMENT,
Id_książki INTEGER,
Id_klienta INTEGER,
CONSTRAINT Id_książki_fk FOREIGN KEY (Id_książki) REFERENCES Półka(Id)
);
